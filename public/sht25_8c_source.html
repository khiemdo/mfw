312 /* ===========================================================================*/
313 /* sht25_run() handles the state machine of sht25 */
314 /* ===========================================================================*/
315 void SHT25_run()
316 {
317  switch(SHT25_state){
318  case E_SHT25_STATE_PENDING:
319  break;
320  case E_SHT25_STATE_IDLE:
321  if(hal_timer_isTimedOut(SHT25_next_cycle)){
322  SHT25_triggerMeasurement();
323  }
324  break;
325  case E_SHT25_STATE_MEASURE:
326  if(hal_timer_isTimedOut(SHT25_measurement_ready)){
327  SHT25_fetchMeasurement();
328  }
329  break;
330  case E_SHT25_STATE_RESET:
331  if(hal_timer_isTimedOut(SHT25_reset_finished)){
332  SHT25_state = E_SHT25_STATE_IDLE;
333  }
334  break;
335  case E_SHT25_STATE_OFF:
336  break;
337  case E_SHT25_STATE_TIMEOUT:
338  break;
339  default:
340  break;
341  }
342 }

347 bool SHT25_get(Humidity *const humidity_msg)
348 {
349  bool b_return = false;
350  if(humidity_msg)
351  {
352  if(SHT25_hasT || SHT25_hasRH)
353  {
354  b_return = true;
355 
356  if(SHT25_hasT)
357  {
358  SHT25_hasT = false;
359  humidity_msg->generated = SHT25_T_time;
360  humidity_msg->has_generated = true;
361  humidity_msg->temperature_raw = (SHT25_T_data[0] << 8) + (SHT25_T_data[1] & 0xFC);
362  humidity_msg->has_temperature_raw = true;
363  }
364 
365  if(SHT25_hasRH)
366  {
367  SHT25_hasRH = false;
368  humidity_msg->generated = SHT25_RH_time;
369  humidity_msg->has_generated = true;
370  humidity_msg->humidity_raw = (SHT25_RH_data[0] << 8) + (SHT25_RH_data[1] & 0xFC);
371  humidity_msg->has_humidity_raw = true;
372  }
373  }
374  }
375  return b_return;
376 }

380 /* ===========================================================================*/
381 void SHT25_setConf(HumidityConf *const humidityConf_msg)
382 {
383  if(humidityConf_msg)
384  {
385  if(humidityConf_msg->has_basic_conf)
386  {
387  if(humidityConf_msg->basic_conf.has_activated)
388  {
389  if(humidityConf_msg->basic_conf.activated)
390  {
391  if(!SHT25_active)
392  {
393  SHT25_state = E_SHT25_STATE_IDLE;
394  SHT25_active = true;
395  }
396  }
397  else
398  {
399  SHT25_active = false; // Deactivate, but finish running measurements
400  }
401  }
402  if(humidityConf_msg->basic_conf.has_rate && humidityConf_msg->basic_conf.rate != 0)
403  {
404  if(humidityConf_msg->basic_conf.rate > 8)
405  {
406  humidityConf_msg->basic_conf.rate = 8;
407  }
408  SHT25_refresh_rate = 1000 / humidityConf_msg->basic_conf.rate;
409  SHT25_cycle_time = SHT25_refresh_rate - (SHT25_measurement_time_RH[SHT25_resolution] + SHT25_measurement_time_T[SHT25_resolution]);
410  }
411  }
412  }

460 void SHT25_triggerMeasurement(void)
461 {

466  case E_SHT25_LAST_T:

482  }

484  {
485  SHT25_state = E_SHT25_STATE_OFF;

523 bool SHT25_checkCRC(SHT25_DATA_t ui16_data, SHT25_CRC_t ui8_crc)
524 {
525  bool b_return = false;
526  SHT25_CRC_t ui8_checksum = 0;
527  uint8_t ui8_high = 0;
528  uint8_t ui8_low = 0;
529  uint8_t ui8_bit = 0;
530 
531  ui8_low = ui16_data & 0x00FF;
532  ui8_high = (ui16_data & 0xFF00) >> 8;
533 
534  ui8_checksum ^= ui8_high;
535  for(ui8_bit = 8; ui8_bit > 0; ui8_bit--)
536  {
537  if(ui8_checksum & 0x80)
538  {
539  ui8_checksum = (ui8_checksum << 1) ^ SHT25_CRC8_POLYNOMIAL;
540  }
541  else
542  {
543  ui8_checksum = (ui8_checksum << 1);
544  }
545  }
546  ui8_checksum ^= ui8_low;
547  for(ui8_bit = 8; ui8_bit > 0; ui8_bit--)
548  {
549  if(ui8_checksum & 0x80)
550  {
551  ui8_checksum = (ui8_checksum << 1) ^ SHT25_CRC8_POLYNOMIAL;
552  }
553  else
554  {
555  ui8_checksum = (ui8_checksum << 1);
556  }
557  }
558 
559  if (ui8_crc == ui8_checksum)
560  {
561  b_return = true;
562  }
563  return b_return;
564 }
