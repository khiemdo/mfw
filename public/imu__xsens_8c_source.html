100 s_imu_xsens_ctx_t* xsens_createImu(XsensConnType connType, uint32_t speed)
101 {
102  s_imu_xsens_ctx_t* ret = NULL;
103  //Reset memory in struct to 0
104  memset( &gs_xsens_imu_ctx, 0, sizeof(gs_xsens_imu_ctx) );
105  //Do initialization of communication port
106  switch (connType){
107  case xsensSPI:
108  break;
109  case xsensI2C:
110  break;
111  case xsensUART:
112  gs_xsens_imu_ctx.uart = hal_uart_init(E_HAL_UART_PORT_0, speed);
113  if (gs_xsens_imu_ctx.uart != NULL){
114  gs_xsens_imu_ctx.connectionType = xsensUART;
115  gs_xsens_imu_ctx.connectionInitialized = true;
116  }
117  break;
118  default:
119  break;
120  }
121 
122  if (gs_xsens_imu_ctx.connectionInitialized)
123  {
124  //Setting up callbacks for Message parser
125  g_queue_data = circular_buffer_init(32);
126  g_queue_response = circular_buffer_init(32);
127  XbusParserCallback xbusCallback = {};
128  xbusCallback.allocateBuffer = loc_allocateMessageData;
129  xbusCallback.deallocateBuffer = loc_deallocateMessageData;
130  xbusCallback.handleMessage = loc_mtMessageHandler;
131  gs_xsens_imu_ctx.xbusParser = XbusParser_create(&xbusCallback);
132  // Initalize Protobuf Imu
133  gs_xsens_imu_ctx.protoData.measurements_count = 0;
134  gs_xsens_imu_ctx.maxMeasurementCount = (uint8_t) sizeof(gs_xsens_imu_ctx.protoData.measurements) / sizeof(Imu);
135  ret = &gs_xsens_imu_ctx;
136  }
137  return ret;
138 }

301 static void loc_mtMessageHandler(XbusMessage const* message)
302 {
303  if (message)
304  {
305  if (message->mid == XMID_MtData2)
306  {
307  ImuMeasurement* measurement = &gs_xsens_imu_ctx.protoData.measurements[gs_xsens_imu_ctx.protoData.measurements_count];
308  uint16_t counter;
309  if (XbusMessage_getDataItem(&counter, XDI_PacketCounter, message))
310  {
311  //measurement->time.has_ticks_measured = true;
312  //measurement->time.ticks_measured.onepps_counter = counter;
313  }
314  uint32_t sampleTime;
315  if (XbusMessage_getDataItem(&sampleTime, XDI_SampleTimeFine, message))
316  {
317  //measurement->time.has_ticks_measured = true;
318  //measurement->time.ticks_measured.ticks = sampleTime;
319  }
320  float ori[4];
321  if (XbusMessage_getDataItem(ori, XDI_Quaternion, message))
322  {
323  measurement->has_quaternion = true;
324  measurement->quaternion.x = ori[0];
325  measurement->quaternion.y = ori[1];
326  measurement->quaternion.z = ori[2];
327  measurement->quaternion.w = ori[3];
328  }
329  float acc[3];
330  if (XbusMessage_getDataItem(acc, XDI_Acceleration, message))
331  {
332  measurement->has_accel = true;
333  measurement->accel.x = acc[0];
334  measurement->accel.y = acc[1];
335  measurement->accel.z = acc[2];
336 
337  }
338  float gyr[3];
339  if (XbusMessage_getDataItem(gyr, XDI_RateOfTurn, message))
340  {
341  measurement->has_gyro = true;
342  measurement->gyro.x = gyr[0];
343  measurement->gyro.y = gyr[1];
344  measurement->gyro.z = gyr[2];
345  }
346  float mag[3];
347  if (XbusMessage_getDataItem(mag, XDI_MagneticField, message))
348  {
349  measurement->has_mag = true;
350  measurement->mag.x = mag[0];
351  measurement->mag.y = mag[1];
352  measurement->mag.z = mag[2];
353  }
354  gs_xsens_imu_ctx.protoData.measurements_count++;
355  //Check for overflow of array and reset if necessary
356  if (gs_xsens_imu_ctx.protoData.measurements_count >= gs_xsens_imu_ctx.maxMeasurementCount){
357  gs_xsens_imu_ctx.protoData.measurements_count = 0;
358  }
359  loc_deallocateMessageData(message->data);
360  }
361  else
362  {
363  XbusMessage* m = malloc(sizeof(XbusMessage));
364  if (m)
365  {
366  memcpy(m, message, sizeof(XbusMessage));
367  circular_buffer_push(g_queue_response, (void*) m);
368  }
369  else if (message->data)
370  {
371  loc_deallocateMessageData(message->data);
372  }
373  }
374  }
375 }

435 static XbusMessage* loc_doTransaction(s_imu_xsens_ctx_t const* cntx, XbusMessage const* m)
436 {
437  uint8_t buffer[READ_SIZE];
438  loc_sendMessage(cntx, m);
439  memset(buffer, 0, sizeof(buffer));
440  XbusMessage* response = NULL;
441  while (response == NULL)
442  {
443  uint16_t bytes_read = 0;
444  switch (cntx->connectionType){
445  case xsensUART:
446  bytes_read = hal_uart_read(cntx->uart,buffer, READ_SIZE);
447  break;
448  case xsensSPI:
449  break;
450  case xsensI2C:
451  break;
452  }
453  if (bytes_read > 0)
454  {
455  XbusParser_parseBuffer(cntx->xbusParser,buffer, bytes_read);
456  response = (XbusMessage*)circular_buffer_pop(g_queue_response);
457  }
458  }
459  return response;
460 }
