128 bool timing_getTimestamp(Timestamp * const timestamp_msg)
129 {
130  bool b_return = false;
131 
132  if(timestamp_msg)
133  {
134  #if TIMING_USE_LOW_PRECISION_TIMER
135  uint32_t currentTimeMs = 0;
136 
137  /* Requesting the timeout for '0' milliseconds actually returns the current
138  * time since last startup. */
139  currentTimeMs = hal_timer_getTimeout(0);
140 
141  /* Convert from milliseconds to seconds. */
142  timestamp_msg->sec = (currentTimeMs/1000);
143  timestamp_msg->has_sec = true;
144 
145  /* Convert from milliseconds to nanoseconds within last second. */
146  timestamp_msg->nsec = ((currentTimeMs % 1000) * 8000);
147  timestamp_msg->has_nsec = true;
148 
149  b_return = true;
150 
151  #elif TIMING_USE_HIGH_PRECISION_TIMER
152  /* High precision timer implementation here. */
153  b_return = hal_pps_getTimestamp(timestamp_msg);
154 
155  #endif
156  }
157 
158  return b_return;
159 }
