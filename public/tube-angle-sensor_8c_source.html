135 static void loc_callbackToElevationIdle(void)
136 {
137  hal_gpio_write(gpio_port_cs0, GPIO_CS0_MASK, GPIO_HIGH);
138  /* Copy the data from array to our local Imu struct variable.
139  * Furthermore, this function creates the timestamp for the measured data!
140  * I.e. the timestamp is now applied right after the last byte from MPU is
141  * received through SPI interface.
142  */
143  el_data_avail = true;
144  loc_copyLastElDataToMsg( &g_lastElData );
145  elevation_state = E_TUBE_ANGLE_SENSOR_ELEVATION_STATE_IDLE;
146  return;
147 }

171 /* servo_dude_init() Initializes a ServoDude and returns context */
172 /* ===========================================================================*/
173 bool tube_angle_sens_init(uint32_t ui32_spiClock) {
174  bool b_return = false;
176  elevation_spi_ctx = hal_spi_init(SPI_PORT_ELEVATION, E_HAL_SPI_MODE_MASTER, ui32_spiClock);
177  rotation_spi_ctx = hal_spi_init(SPI_PORT_ROTATION, E_HAL_SPI_MODE_MASTER, ui32_spiClock);
178  elevation_state = E_TUBE_ANGLE_SENSOR_ELEVATION_STATE_IDLE;
179  rotation_state = E_TUBE_ANGLE_SENSOR_ROTATION_STATE_IDLE;
180  elevation_rate = hal_timer_getTimeout(TAS_REFRESH_PERIOD);
181  rotation_rate = hal_timer_getTimeout(TAS_REFRESH_PERIOD);
183  b_return = hal_spi_config(elevation_spi_ctx, E_HAL_SPI_CLK_POL_LOW, E_HAL_SPI_CLK_PHA_EDGE_SCND, ui32_spiClock);
184  if (!b_return) return false;
185  b_return = hal_spi_config(rotation_spi_ctx, E_HAL_SPI_CLK_POL_LOW, E_HAL_SPI_CLK_PHA_EDGE_SCND, ui32_spiClock);
186  if (!b_return) return false;
187 
189  b_return = hal_gpio_init();
190  b_return = hal_gpio_setDir(GPIO_CS0_PORT, GPIO_CS0_MASK, E_HAL_GPIO_DIR_OUT);
191  b_return = hal_gpio_setDir(GPIO_CS1_PORT, GPIO_CS1_MASK, E_HAL_GPIO_DIR_OUT);
192  gpio_port_cs0 = GPIO_CS0_PORT;
193  gpio_port_cs1 = GPIO_CS1_PORT;
194 
196  b_return = hal_gpio_write(gpio_port_cs0, GPIO_CS0_MASK, GPIO_HIGH);
197  b_return = hal_gpio_write(gpio_port_cs1, GPIO_CS1_MASK, GPIO_HIGH);
198 
200  g_lastElData = g_AngleEncoderDefault;
201  g_lastRoData = g_AngleEncoderDefault;
202 
203  el_data_avail = false;
204  ro_data_avail = false;
205 
206  #if TUBE_ANGLE_ENABLE_AFTER_START
207  gb_tasEnabled = true;
208  #else
209 
211  gb_tasEnabled = false;
212  #endif
213 
214  return (b_return);
215 }

223 void tube_angle_sens_run() {
224 
225  if( gb_tasEnabled )
226  {
228  switch(elevation_state){
229  case E_TUBE_ANGLE_SENSOR_ELEVATION_STATE_IDLE:
230  if (hal_timer_isTimedOut(elevation_rate)) {
231  elevation_rate = hal_timer_getTimeout(TAS_REFRESH_PERIOD);
232  elevation_timeout = hal_timer_getTimeout(TIMEOUT_DELAY);
233  elevation_state = E_TUBE_ANGLE_SENSOR_ELEVATION_STATE_PENDING;
234  hal_gpio_write(gpio_port_cs0, GPIO_CS0_MASK, GPIO_LOW);
235  hal_spi_xfer(elevation_spi_ctx, tas_el_data_dummy, tas_el_data, 2, loc_callbackToElevationIdle);
236  }
237  break;
238  case E_TUBE_ANGLE_SENSOR_ELEVATION_STATE_PENDING:
239  if (hal_timer_isTimedOut(elevation_timeout)) {
240  elevation_state = E_TUBE_ANGLE_SENSOR_ELEVATION_STATE_TIMEOUT;
241  }
242  break;
243  case E_TUBE_ANGLE_SENSOR_ELEVATION_STATE_TIMEOUT:
244  break;
245  }
247  switch(rotation_state){
248  case E_TUBE_ANGLE_SENSOR_ROTATION_STATE_IDLE:
249  if (hal_timer_isTimedOut(rotation_rate)) {
250  rotation_rate = hal_timer_getTimeout(TAS_REFRESH_PERIOD);
251  rotation_timeout = hal_timer_getTimeout(TIMEOUT_DELAY);
252  rotation_state = E_TUBE_ANGLE_SENSOR_ROTATION_STATE_PENDING;
253  hal_gpio_write(gpio_port_cs1, GPIO_CS1_MASK, GPIO_LOW);
254  hal_spi_xfer(rotation_spi_ctx, tas_ro_data_dummy, tas_ro_data, 2, loc_callbackToRotationIdle);
255  }
256  break;
257  case E_TUBE_ANGLE_SENSOR_ROTATION_STATE_PENDING:
258  if (hal_timer_isTimedOut(rotation_timeout)) {
259  rotation_state = E_TUBE_ANGLE_SENSOR_ROTATION_STATE_TIMEOUT;
260  }
261  break;
262  case E_TUBE_ANGLE_SENSOR_ROTATION_STATE_TIMEOUT:
263  break;
264  }
265  }
266 
267  return;
268 }
