269 void ads1115_run(s_ads1115_ctx_t * ads_ctx) {
271  i2c_bus_mgr_run();
272  E_I2C_BUS_NODE_T i2c_node;
273  i2c_node = ads_ctx->ads_id;
275  switch(ads_ctx->ads_id){
276  case E_I2C_BUS_NODE_ADC1:
277  write_ctx1.slave_address = ads_ctx->ads_slave_address;
278  read_ctx1.slave_address = ads_ctx->ads_slave_address;
279  write_ctx1.write_data = ads_ctx->ads_tx_data;
280  read_ctx1.write_data = ads_ctx->ads_tx_data;
281  read_ctx1.read_data = ads_ctx->ads_rx_data;
282  break;
283  case E_I2C_BUS_NODE_ADC2:
284  write_ctx2.slave_address = ads_ctx->ads_slave_address;
285  read_ctx2.slave_address = ads_ctx->ads_slave_address;
286  write_ctx2.write_data = ads_ctx->ads_tx_data;
287  read_ctx2.write_data = ads_ctx->ads_tx_data;
288  read_ctx2.read_data = ads_ctx->ads_rx_data;
289  break;
290  case E_I2C_BUS_NODE_ADC3:
291  write_ctx3.slave_address = ads_ctx->ads_slave_address;
292  read_ctx3.slave_address = ads_ctx->ads_slave_address;
293  write_ctx3.write_data = ads_ctx->ads_tx_data;
294  read_ctx3.write_data = ads_ctx->ads_tx_data;
295  read_ctx3.read_data = ads_ctx->ads_rx_data;
296  break;
297  }
305  switch(ads_ctx->ads_state){
306  case E_ADS1115_STATE_IDLE:
307  ads_ctx->ads_tx_data[0] = 0x01; // Write to config register
308  ads_ctx->ads_tx_data[1] = 0xC5 | ((ads_ctx->ads_channel & 0x03) << 4); // Start single conv, channel 0, +-2.048 V, single shot (Select channel here!)
309  ads_ctx->ads_tx_data[2] = 0xE3; // 860 SPS, Trad. comp, active low comp, no latch, disable comp.
310  switch(ads_ctx->ads_id){
311  case E_I2C_BUS_NODE_ADC1:
312  write_ctx1.write_length = 3;
313  write_ctx1.callback_pointer = &loc_ads1115_id1_callback_to_ready;
314  ads_ctx->ads_state = E_ADS1115_STATE_PENDING;
315  gs_ads1115_id2.timeout_timer = hal_timer_getTimeout(TIMEOUT_DELAY);
316  i2c_bus_mgr_queue_write(i2c_node,&write_ctx1);
317  break;
318  case E_I2C_BUS_NODE_ADC2:
319  write_ctx2.write_length = 3;
320  write_ctx2.callback_pointer = &loc_ads1115_id2_callback_to_ready;
321  ads_ctx->ads_state = E_ADS1115_STATE_PENDING;
322  gs_ads1115_id2.timeout_timer = hal_timer_getTimeout(TIMEOUT_DELAY);
323  i2c_bus_mgr_queue_write(i2c_node,&write_ctx2);
324  break;
325  case E_I2C_BUS_NODE_ADC3:
326  write_ctx3.write_length = 3;
327  write_ctx3.callback_pointer = &loc_ads1115_id3_callback_to_ready;
328  ads_ctx->ads_state = E_ADS1115_STATE_PENDING;
329  gs_ads1115_id2.timeout_timer = hal_timer_getTimeout(TIMEOUT_DELAY);
330  i2c_bus_mgr_queue_write(i2c_node,&write_ctx3);
331  break;
332  }
333 
334  break;
335  case E_ADS1115_STATE_PENDING:
336  if (hal_timer_isTimedOut(ads_ctx->timeout_timer)) {
337  //ads_ctx->ads_state = E_ADS1115_STATE_TIMEOUT;
338  }
339  break;
340  case E_ADS1115_STATE_MEASURING:
341  if (hal_timer_isTimedOut(ads_ctx->measure_timeout)) {
342  ads_ctx->ads_state = E_ADS1115_STATE_READY;
343  }
344  break;
345  case E_ADS1115_STATE_READY:
346  ads_ctx->ads_tx_data[0] = 0x00;
347  read_ctx1.write_length = 1;
348  read_ctx2.write_length = 1;
349  read_ctx3.write_length = 1;
350  ads_ctx->ads_rx_data[0] = 0x00;
351  ads_ctx->ads_rx_data[1] = 0x00;
352  ads_ctx->ads_rx_data[2] = 0x00;
353  read_ctx1.read_length = 2;
354  read_ctx2.read_length = 2;
355  read_ctx3.read_length = 2;
356  switch(ads_ctx->ads_id){
357  case E_I2C_BUS_NODE_ADC1:
358  read_ctx1.callback_pointer = &loc_ads1115_id1_callback_to_idle;
359  ads_ctx->ads_state = E_ADS1115_STATE_PENDING;
360  gs_ads1115_id2.timeout_timer = hal_timer_getTimeout(TIMEOUT_DELAY);
361  i2c_bus_mgr_queue_read(i2c_node,&read_ctx1);
362  break;
363  case E_I2C_BUS_NODE_ADC2:
364  read_ctx2.callback_pointer = &loc_ads1115_id2_callback_to_idle;
365  ads_ctx->ads_state = E_ADS1115_STATE_PENDING;
366  gs_ads1115_id2.timeout_timer = hal_timer_getTimeout(TIMEOUT_DELAY);
367  i2c_bus_mgr_queue_read(i2c_node,&read_ctx2);
368  break;
369  case E_I2C_BUS_NODE_ADC3:
370  read_ctx3.callback_pointer = &loc_ads1115_id3_callback_to_idle;
371  ads_ctx->ads_state = E_ADS1115_STATE_PENDING;
372  gs_ads1115_id2.timeout_timer = hal_timer_getTimeout(TIMEOUT_DELAY);
373  i2c_bus_mgr_queue_read(i2c_node,&read_ctx3);
374  break;
375  }
376  break;
377  case E_ADS1115_STATE_TIMEOUT:
