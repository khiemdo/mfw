148 /* ===========================================================================*/
149 static s_hal_i2c_ctx_t *loc_initI2C0(uint32_t ui32_sysClock)
150 {
151 
152  // Enable the peripherals used by this example.
153  //
154  SysCtlPeripheralEnable(SYSCTL_PERIPH_I2C0);
155  SysCtlPeripheralEnable(SYSCTL_PERIPH_GPIOB);
156 
157  //
158  // Set GPIO B2 and B3 as I2C pins.
159  //
160  GPIOPinConfigure(GPIO_PB2_I2C0SCL);
161  GPIOPinConfigure(GPIO_PB3_I2C0SDA);
162  GPIOPinTypeI2CSCL(GPIO_PORTB_BASE, GPIO_PIN_2);
163  GPIOPinTypeI2C(GPIO_PORTB_BASE, GPIO_PIN_3);
164 
165  //
166  // init i2c clock
167  //
168  I2CMasterInitExpClk(I2C0_BASE, ui32_sysClock, true);
169 
170  /* Initialize our context struct. */
171  memset( &gs_i2c0_ctx, 0, sizeof(gs_i2c0_ctx) );
172  gs_i2c0_ctx.state = E_HAL_I2C_STATE_IDLE;
173  memset( &gs_i2c0_inst, 0, sizeof(gs_i2c0_inst) );
174  I2CMInit(&gs_i2c0_inst, I2C0_BASE, INT_I2C0, 0xFF, 0xFF,ui32_sysClock);
175  gs_i2c0_ctx.gs_i2c_inst = & gs_i2c0_inst;
176 
177  /* Finally, register our interrupt handler implementation to the MCU here.
178  * When interrupt occurs, this function gets called and forwards the call
179  * to I2CMIntHandler() that is provided by i2cm_drv.h */
180  I2CIntRegister(I2C0_BASE, IntI2C0Handler);
181 
182  return &gs_i2c0_ctx;
183 }

385 /* hal_i2c_init() Initializes a I2C Port and returns context */
386 /* ===========================================================================*/
387 s_hal_i2c_ctx_t *hal_i2c_ti_drv_init(E_HAL_I2C_PORT_t e_port)
388 {
389  s_hal_i2c_ctx_t *ps_return = NULL;
390 
391  /* Get the System clock (settings). */
392  uint32_t ui32SysClock = hal_mcu_getSysClock();
393 
394  switch (e_port)
395  {
396  case E_HAL_I2C_PORT_0:
397  ps_return = loc_initI2C0(ui32SysClock);
398  break;
399 
400  case E_HAL_I2C_PORT_1:
401  ps_return = loc_initI2C1(ui32SysClock);
402  break;
403 
404  case E_HAL_I2C_PORT_2:
405  ps_return = loc_initI2C2(ui32SysClock);
406  break;
407 
408  case E_HAL_I2C_PORT_3:
409  ps_return = loc_initI2C3(ui32SysClock);
410  break;
411 
412  case E_HAL_I2C_PORT_4:
413  ps_return = loc_initI2C4(ui32SysClock);
414  break;
415 
416  case E_HAL_I2C_PORT_5:
417  ps_return = loc_initI2C5(ui32SysClock);
418  break;
419 
420  /* Other I2C ports are not supported => ps_return remains NULL. */
421  default:
422  break;
423  }
424 
425  return ps_return;
426 }
