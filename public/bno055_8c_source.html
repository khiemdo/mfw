270 /* loc_setOperationMode() */

272 bool loc_setOperationMode(E_BNO055_MODE_t mode) {
273  bool b_return = false;
274  if ((ge_bno_state == E_BNO055_STATE_IDLE_WRITE)||(ge_bno_state == E_BNO055_STATE_IDLE_READ)) {
275  ge_operation_mode = mode;
276  ge_bno_state = E_BNO055_STATE_INIT_PENDING;
277  uint8_t data[2];
278  data[0] = REG_OPR_MODE;
279  switch(mode) {
280  case E_BNO055_MODE_NORMAL:
281  data[1] = OPR_MODE_AMG;
282  break;
283  case E_BNO055_MODE_FUSION:
284  data[1] = OPR_MODE_NDOF;
285  break;
286  case E_BNO055_MODE_ALL:
287  data[1] = OPR_MODE_NDOF;
288  break;
289  default:
290  data[1] = OPR_MODE_AMG;
291  break;
292  }
293  BNO055_bno_timeout = hal_timer_getTimeout(BNO_TIMEOUT_DELAY);
294  BNO055_update_timer = hal_timer_getTimeout(BNO_REFRESH_PERIOD);
295  hal_i2c_write(bno055_i2c_ctx, SLAVE_ADDRESS, data,2, loc_bno_callbackToIdle);
296 
297  // Needed for init maybe better solution later!
298  __delay_cycles (1000000);
299 
300  b_return = true;
301  } else {
302  b_return = false;
303  }
304  return b_return;
305 }

314 bool bno055_init() {
315  bool b_return = false;
316 
317  g_bno_lastData = g_bno_imuDefault;
318  /* Initialize our context struct and init I2C port. */
319  ge_bno_state = E_BNO055_STATE_IDLE_WRITE;
320  bno055_i2c_ctx = hal_i2c_ti_drv_init(BNO055_I2C_PORT);
321  if (bno055_i2c_ctx != NULL) b_return = true;
322 
323 
324  if(b_return)
325  {
326  b_return = hal_gpio_setDir(E_HAL_GPIO_PORT_Q, HAL_GPIO_PIN_4, E_HAL_GPIO_DIR_OUT);
327  }
328  if(b_return)
329  {
330  b_return = hal_gpio_write(E_HAL_GPIO_PORT_Q, HAL_GPIO_PIN_4, HAL_GPIO_PIN_4);
331  }
332  if(b_return)
333  {
334  b_return = hal_gpio_setDir(E_HAL_GPIO_PORT_N, HAL_GPIO_PIN_2, E_HAL_GPIO_DIR_OUT);
335  }
336  if(b_return)
337  {
338  b_return = hal_gpio_write(E_HAL_GPIO_PORT_N, HAL_GPIO_PIN_2, HAL_GPIO_PIN_2);
339  }
340 
341  if(b_return)
342  {
343  b_return = hal_gpio_write(E_HAL_GPIO_PORT_Q, HAL_GPIO_PIN_4, ~HAL_GPIO_PIN_4);
344  __delay_cycles(10);
345  }
346  if(b_return)
347  {
348  b_return = hal_gpio_write(E_HAL_GPIO_PORT_Q, HAL_GPIO_PIN_4, HAL_GPIO_PIN_4);
349  }
350  __delay_cycles(78000000);
351 
352 
353 
354 
355  /* Set the operation mode of the bno055. */
356  loc_setOperationMode(E_BNO055_MODE_ALL);
357 
358  gb_new_data_av = false;
359 
360  #if BNO055_ENABLE_AFTER_START
361  gb_bnoEnabled = true;
362  #else
363 
365  gb_bnoEnabled = false;
366  #endif
367  /* By default this module is disabled after init(). It must be enabled
368  * by a call to \ref setConf() with corresponding member set to true. */
369  BNO055_update_timer = hal_timer_getTimeout(BNO_STARTUP_DELAY);
370 
371  return (b_return);
372 }

378 void bno055_run() {
379  if( gb_bnoEnabled ) {
380  uint8_t read_addr;
381  uint8_t read_len;
382  switch (ge_operation_mode) {
383  case E_BNO055_MODE_NORMAL:
384  read_addr = REG_ACC_DATA_X_LSB;
385  read_len = 18;
386  break;
387  case E_BNO055_MODE_FUSION:
388  read_addr = REG_QUA_DATA_W_LSB;
389  read_len = 20;
390  break;
391  case E_BNO055_MODE_ALL:
392  read_addr = REG_ACC_DATA_X_LSB;
393  read_len = 45;
394  break;
395  default:
396  read_addr = REG_ACC_DATA_X_LSB;
397  read_len = 18;
398  break;
399  }
400  switch(ge_bno_state){
401  case E_BNO055_STATE_IDLE_WRITE:
402  if (hal_timer_isTimedOut(BNO055_update_timer)) {
403  BNO055_update_timer = hal_timer_getTimeout(BNO_REFRESH_PERIOD);
404  BNO055_bno_timeout = hal_timer_getTimeout(BNO_TIMEOUT_DELAY);
405  ge_bno_state = E_BNO055_STATE_PENDING_WRITE;
406  hal_i2c_write(bno055_i2c_ctx, SLAVE_ADDRESS, &read_addr, 1, loc_bno_callbackToIdleRead);
407  }
408  break;
409  case E_BNO055_STATE_IDLE_READ:
410  ge_bno_state = E_BNO055_STATE_PENDING_READ;
411  hal_i2c_read(bno055_i2c_ctx, SLAVE_ADDRESS, &read_addr, 0, rx_data, read_len, loc_bno_callbackToIdleWrite);
412  break;
413  case E_BNO055_STATE_INIT_PENDING:
414  if (hal_timer_isTimedOut(BNO055_bno_timeout)) {
415  ge_bno_state = E_BNO055_STATE_TIMEOUT;
416  }
417  break;
418  case E_BNO055_STATE_PENDING_WRITE:
419  if (hal_timer_isTimedOut(BNO055_bno_timeout)) {
420  ge_bno_state = E_BNO055_STATE_TIMEOUT;
421  }
422  break;
423  case E_BNO055_STATE_PENDING_READ:
424  if (hal_timer_isTimedOut(BNO055_bno_timeout)) {
425  ge_bno_state = E_BNO055_STATE_TIMEOUT;
426  }
427  break;
428  case E_BNO055_STATE_DATA_READY:
429  if (hal_timer_isTimedOut(BNO055_bno_timeout)) {
430  ge_bno_state = E_BNO055_STATE_TIMEOUT;
431  }
432  break;
433  case E_BNO055_STATE_TIMEOUT:
434  ge_bno_state = E_BNO055_STATE_TIMEOUT;
435  break;
436  default:
437  break;
438  }
439  }
440  return;
441 }
