146 void pps_run()
147 {
148  switch(PPS_state)
149  {
150  case E_PPS_STATE_IDLE:
151  if(hal_timer_isTimedOut(PPS_timer) && current_mode == PpsConf_Mode_FC_TRIG_MODE)
152  {
153  PPS_timer = hal_timer_getTimeout(PPS_TIMER_LENGTH);
154  loc_generate_pps_tick();
155  }
156  if(hal_timer_isTimedOut(PPS_timeout))
157  {
158  current_mode = last_mode = PpsConf_Mode_FC_TRIG_MODE;
159  hal_gpio_setDir(MFW_PPS_INPUT_PORT, MFW_PPS_INPUT_PIN, E_HAL_GPIO_DIR_OUT);
160  }
161  if(current_mode != PpsConf_Mode_GPS_TRIG_MODE)
162  {
163  hal_gpio_write(MFW_PPS_INPUT_PORT, MFW_PPS_INPUT_PIN, 0x0);
164  }
165  #if MFW_PPS_LED_EN
166  if(hal_timer_isTimedOut(PPS_led_on))
167  {
168  hal_gpio_write(MFW_PPS_LED_PORT, MFW_PPS_LED_PIN, 0x0);
169  }
170  #endif
171  break;
172  case E_PPS_STATE_DATA_READY:
173  break;
174  default:
175  break;
176  }
177  return;
178 }

193 bool pps_setConf(PpsConf *const ppsConf_msg)
194 {
195  bool b_return = false;
196 
197  if(ppsConf_msg->has_mode)
198  {
199  current_mode = ppsConf_msg->mode;
200 
201  if(current_mode == last_mode)
202  {
203  if(current_mode == PpsConf_Mode_GS_TRIG_MODE)
204  {
205  b_return = loc_generate_pps_tick();
206  }
207  else
208  {
209  b_return = true;
210  }
211  }
212  else
213  {
214  switch(current_mode)
215  {
216  case PpsConf_Mode_FC_TRIG_MODE:
217  b_return = hal_gpio_setDir(MFW_PPS_INPUT_PORT, MFW_PPS_INPUT_PIN, E_HAL_GPIO_DIR_OUT);
218  break;
219  case PpsConf_Mode_GS_TRIG_MODE:
220  b_return = hal_gpio_setDir(MFW_PPS_INPUT_PORT, MFW_PPS_INPUT_PIN, E_HAL_GPIO_DIR_OUT);
221  if(b_return)
222  {
223  b_return = loc_generate_pps_tick();
224  }
225  break;
226  case PpsConf_Mode_GPS_TRIG_MODE:
227  b_return = hal_gpio_setDir(MFW_PPS_INPUT_PORT, MFW_PPS_INPUT_PIN, E_HAL_GPIO_DIR_IN);
228  break;
229  default:
230  break;
231  }
232  }
233 
234  last_mode = current_mode;
235  }
236 
237  return b_return;
238 }
