102 static void loc_mpu_callbackToIdle(void)
103 {
104  /* Pull up CS pin to finalize SPI transmission. */
105  hal_gpio_write(MPU9250_gpio_port, GPIO_CS_MASK, GPIO_HIGH);
106 
107  /* Copy the data from array to our local Imu struct variable.
108  * Furthermore, this function creates the timestamp for the measured data!
109  * I.e. the timestamp is now applied right after the last byte from MPU is
110  * received through SPI interface. Makes also sure, that no MPU_get is in
111  * progress.
112  */
113  loc_copyLastDataToMsg(&MPU9250_lastData);
114  MPU9250_isDataReady = true;
115 
116  MPU9250_state = E_MPU9250_STATE_DATA_READY;
117  return;
118 }

120 static void loc_copyLastDataToMsg(Imu *const mpu_msg)
121 {
122 
123  mpu_msg->has_generated = hal_pps_getTimestamp(&(mpu_msg->generated));
124 
125  // get accel value
126  mpu_msg->accel_raw.x = (int16_t)((MPU9250_meas_data[1] << 8) | (MPU9250_meas_data[2]));
127  mpu_msg->accel_raw.has_x = true;
128  mpu_msg->accel_raw.y = (int16_t)((MPU9250_meas_data[3] << 8) | (MPU9250_meas_data[4]));
129  mpu_msg->accel_raw.has_y = true;
130  mpu_msg->accel_raw.z = (int16_t)((MPU9250_meas_data[5] << 8) | (MPU9250_meas_data[6]));
131  mpu_msg->accel_raw.has_z = true;
132  mpu_msg->has_accel_raw = true;
133 
134  mpu_msg->gyro_raw.x = (int16_t)((MPU9250_meas_data[9] << 8) | (MPU9250_meas_data[10]));
135  mpu_msg->gyro_raw.has_x = true;
136  mpu_msg->gyro_raw.y = (int16_t)((MPU9250_meas_data[11] << 8) | (MPU9250_meas_data[12]));
137  mpu_msg->gyro_raw.has_y = true;
138  mpu_msg->gyro_raw.z = (int16_t)((MPU9250_meas_data[13] << 8) | (MPU9250_meas_data[14]));
139  mpu_msg->gyro_raw.has_z = true;
140  mpu_msg->has_gyro_raw = true;

148 /* mpu9250_init() Initializes a MPU9250 and returns context */
149 /* ===========================================================================*/
150 bool mpu9250_init(E_HAL_SPI_PORT_t e_hal_spi_port, uint32_t ui32_spiClock)
151 {
152  bool b_return = false;
153 
154  MPU9250_lastData = g_mpu_imuDefault;
155  MPU9250_isDataReady = false;
156 
157  memset((void *)MPU9250_meas_data, 0, sizeof(MPU9250_meas_data));
158  memset((void *)MPU9250_addr_data, 0, sizeof(MPU9250_addr_data));
159  MPU9250_addr_data[0] = (MPU9250_REG_MPU_READ_FLAG | MPU9250_REG_ACCEL_XOUT_H);
160 
161  MPU9250_spi_ctx = hal_spi_init(e_hal_spi_port, E_HAL_SPI_MODE_MASTER, ui32_spiClock);
162 
163  /* hal_spi_init() may return a NULL pointer if init fails! */
164  if(MPU9250_spi_ctx)
165  {
166  // Configure the SPI port
167  b_return = hal_spi_config(MPU9250_spi_ctx, E_HAL_SPI_CLK_POL_HIGH, E_HAL_SPI_CLK_PHA_EDGE_SCND, ui32_spiClock);
168 
169  /* If initialization of SPI fails we should not continue. */
170  if(b_return)
171  {
172  // initialize GPIO module with GPIO-pin as CS
173  b_return = hal_gpio_init();
174  b_return = hal_gpio_setDir(GPIO_CS_PORT, GPIO_CS_MASK, E_HAL_GPIO_DIR_OUT);
175  MPU9250_gpio_port = GPIO_CS_PORT;
176 
177  /* If we were not able to pull down the CS pin we can not wake up the MPU. */
178  if(b_return)
179  {
180  // Wake up the imu
181  MPU9250_wake_up_data[0] = MPU9250_REG_PWR_MGMT_1;
182  MPU9250_wake_up_data[1] = 0x00;
183  MPU9250_mpu_timeout = hal_timer_getTimeout(TIMEOUT_DELAY);
184  MPU9250_update_timer = hal_timer_getTimeout(MPU_REFRESH_PERIOD);
185  MPU9250_state = E_MPU9250_STATE_INIT_PENDING;
186  hal_gpio_write(MPU9250_gpio_port, GPIO_CS_MASK, GPIO_LOW);
187  hal_spi_xfer(MPU9250_spi_ctx, MPU9250_wake_up_data, MPU9250_wake_up_dataRX, 2, loc_mpu_callbackToIdle);
188  }
189  }
190  }
191 
192  return b_return;
193 }

199 void mpu9250_run()
200 {
201  switch(MPU9250_state)
202  {
203  case E_MPU9250_STATE_IDLE:
204  if(hal_timer_isTimedOut(MPU9250_update_timer))
205  {
206  MPU9250_update_timer = hal_timer_getTimeout(MPU_REFRESH_PERIOD);
207  MPU9250_mpu_timeout = hal_timer_getTimeout(TIMEOUT_DELAY);
208  MPU9250_state = E_MPU9250_STATE_READ_PENDING;
209  hal_gpio_write(MPU9250_gpio_port, GPIO_CS_MASK, GPIO_LOW);
210  hal_spi_xfer(MPU9250_spi_ctx, (uint8_t *)MPU9250_addr_data, (uint8_t *)MPU9250_meas_data, 15,
211  loc_mpu_callbackToIdle);
212  }
213  break;
214  case E_MPU9250_STATE_READ_PENDING:
215  if(hal_timer_isTimedOut(MPU9250_mpu_timeout))
216  {
217  MPU9250_state = E_MPU9250_STATE_TIMEOUT;
218  }
219  break;
220  case E_MPU9250_STATE_INIT_PENDING:
221  if(hal_timer_isTimedOut(MPU9250_mpu_timeout))
222  {
223  MPU9250_state = E_MPU9250_STATE_TIMEOUT;
224  }
225  break;
226  case E_MPU9250_STATE_DATA_READY:
227  if(hal_timer_isTimedOut(MPU9250_mpu_timeout))
228  {
229  MPU9250_state = E_MPU9250_STATE_TIMEOUT;
230  }
231  break;
232  default:
233  break;
234  }
235 }

251  }
252 
253  return b_return;
