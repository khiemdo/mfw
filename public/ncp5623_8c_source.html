188 /* ===========================================================================*/
189 /* NCP5623_run() handles the state machine of NCP5623 */
190 /* ===========================================================================*/
191 void NCP5623_run()
192 {
193  switch(NCP5623_state){
194  case E_NCP5623_STATE_PENDING:
195  break;
196  case E_NCP5623_STATE_DIMMING:
197  if(hal_timer_isTimedOut(NCP5623_dimming_finished)){
198  NCP5623_state = E_NCP5623_STATE_IDLE;
199  }
200  break;
201  case E_NCP5623_STATE_IDLE:
202  if(hal_timer_isTimedOut(NCP5623_rainbow_timer))
203  NCP5623_rainbow(NCP5623_rainbow_speed);
204  NCP5623_set();
205  break;
206  case E_NCP5623_STATE_TIMEOUT:
207  break;
208  default:
209  break;
210  }

224 void NCP5623_set()
225 {
226  bool b_stay = true;
227  //uint8_t data[1];
228  while(b_stay)
229  {
230  switch(NCP5623_last){
231  case E_NCP5623_LAST_DIMM_TIME:
232  NCP5623_last = E_NCP5623_LAST_BLUE_PWM;
233  if(NCP5623_data.B != NCP5623_data_new.B){
234  NCP5623_state = E_NCP5623_STATE_PENDING;
235  b_stay = false;
236  NCP5623_write_data[0] = NCP5623_CMD_RED_PWM | ((NCP5623_data_new.B >> 3) & NCP5623_CMD_MASK); //Red and Blue are swapped
237  NCP5623_data.B = NCP5623_data_new.B;
238  i2c_bus_mgr_queue_write(E_I2C_BUS_NODE_LED_DRV, &NCP5623_bm_write);
239  //hal_i2c_write(ncp5623_i2c_ctx, NCP5623_addr, data, 1, &loc_NCP5623_cb);
240  }
241  break;
242  case E_NCP5623_LAST_BLUE_PWM:
243  NCP5623_last = E_NCP5623_LAST_GREEN_PWM;
244  if(NCP5623_data.G != NCP5623_data_new.G){
245  NCP5623_state = E_NCP5623_STATE_PENDING;
246  b_stay = false;
247  NCP5623_write_data[0] = NCP5623_CMD_GREEN_PWM | ((NCP5623_data_new.G >> 3) & NCP5623_CMD_MASK);
248  NCP5623_data.G = NCP5623_data_new.G;
249  i2c_bus_mgr_queue_write(E_I2C_BUS_NODE_LED_DRV, &NCP5623_bm_write);

251  }
252  break;
253  case E_NCP5623_LAST_GREEN_PWM:
254  NCP5623_last = E_NCP5623_LAST_RED_PWM;
255  if(NCP5623_data.R != NCP5623_data_new.R){
256  NCP5623_state = E_NCP5623_STATE_PENDING;
257  b_stay = false;
258  NCP5623_write_data[0] = NCP5623_CMD_BLUE_PWM | ((NCP5623_data_new.R >> 3) & NCP5623_CMD_MASK); //Red and Blue are swapped
259  NCP5623_data.R = NCP5623_data_new.R;

261  //hal_i2c_write(ncp5623_i2c_ctx, NCP5623_addr, data, 1, &loc_NCP5623_cb);
262  }
263  break;
264  case E_NCP5623_LAST_RED_PWM:
265  NCP5623_last = E_NCP5623_LAST_DIMM_TIME;
266  b_stay = false;
267  if(NCP5623_data.ILED != NCP5623_data_new.ILED){
268  if(NCP5623_data_new.time == 0)
269  {
270  NCP5623_state = E_NCP5623_STATE_PENDING;
271  NCP5623_write_data[0] = NCP5623_CMD_ILED | (NCP5623_data_new.ILED & NCP5623_CMD_MASK);
272  NCP5623_data.ILED = NCP5623_data_new.ILED;
273  NCP5623_data.time = NCP5623_data_new.time;
274  i2c_bus_mgr_queue_write(E_I2C_BUS_NODE_LED_DRV, &NCP5623_bm_write);
275  //hal_i2c_write(ncp5623_i2c_ctx, NCP5623_addr, data, 1, &loc_NCP5623_cb);
276  }
277  else
278  {
279  NCP5623_last = E_NCP5623_LAST_ILED;
280  NCP5623_state = E_NCP5623_STATE_PENDING;
281  if(NCP5623_data_new.ILED > NCP5623_data.ILED)
282  {
283  NCP5623_write_data[0] = NCP5623_CMD_DIMM_UP | (NCP5623_data_new.ILED & NCP5623_CMD_MASK);
284  NCP5623_data.ILED = NCP5623_data_new.ILED;
285  i2c_bus_mgr_queue_write(E_I2C_BUS_NODE_LED_DRV, &NCP5623_bm_write);

291  NCP5623_data.ILED = NCP5623_data_new.ILED;
292  i2c_bus_mgr_queue_write(E_I2C_BUS_NODE_LED_DRV, &NCP5623_bm_write);

320  NCP5623_rainbow_timer = hal_timer_getTimeout(ui16_speed);

322  {
323  case E_NCP5623_RAINBOW_P1:
