100 void loc_mpl_callback_to_idle(void *pvData, uint_fast8_t ui8Status)
101 {
102  barometer_msg.has_generated = hal_pps_getTimestamp( &(barometer_msg.generated) );
103  barometer_msg.pressure_raw = (mpl_rx_data[0] << 16) | (mpl_rx_data[1] << 8) | (mpl_rx_data[2]);
104  barometer_msg.has_pressure_raw = true;
105  barometer_msg.temperature_raw = (mpl_rx_data[3] << 8) | (mpl_rx_data[4]);
106  barometer_msg.has_temperature_raw = true;
107  mpl_new_data_av = true;
108  mpl_state = E_MPL_STATE_IDLE;
109  i2c_bus_mgr_return_access(E_I2C_BUS_NODE_PRESSURE);
110  mpl_i2c_ctx->state = E_HAL_I2C_STATE_IDLE;
111 }

116 bool MPL_init() {
117  bool b_return = false;
118  mpl_i2c_ctx = hal_i2c_ti_drv_init(E_HAL_I2C_PORT_5);
119  if (mpl_i2c_ctx) {
120  memset(&mpl_write_ctx, 0, sizeof(mpl_write_ctx));
121  mpl_write_ctx.ps_ctx = mpl_i2c_ctx;
122  memset(&mpl_read_ctx, 0, sizeof(mpl_read_ctx));
123  mpl_read_ctx.ps_ctx = mpl_i2c_ctx;
124  b_return = i2c_bus_mgr_init();
125  mpl_state = E_MPL_STATE_INIT1;
126  mpl_new_data_av = false;
127  mpl_i2c_node = E_I2C_BUS_NODE_PRESSURE;
128  mpl_write_ctx.slave_address = MPL_SLAVE_ADDR;
129  mpl_read_ctx.slave_address = MPL_SLAVE_ADDR;
130  mpl_write_ctx.write_data = mpl_tx_data;
131  mpl_read_ctx.write_data = mpl_tx_data;
132  mpl_read_ctx.read_data = mpl_rx_data;
133  }
134 
135  mpl_tx_data[0] = MPL_REG_CTRL1;
136  // Set to Barometer with an OSR = 128
137  mpl_tx_data[1] = 0x38;
138  mpl_write_ctx.write_length = 2;
139  mpl_write_ctx.callback_pointer = &loc_mpl_callback_to_init1;
140  mpl_state = E_MPL_STATE_PENDING;
141  mpl_timeout_timer = hal_timer_getTimeout(TIMEOUT_DELAY);
142  i2c_bus_mgr_queue_write(mpl_i2c_node, &mpl_write_ctx);
143  i2c_bus_mgr_run();
144  return b_return;
145 }

147 void MPL_run() {
148  i2c_bus_mgr_run();
149 
150  switch(mpl_state){
151  case E_MPL_STATE_INIT1:
152  mpl_tx_data[0] = MPL_REG_DATA_CFG;
153  // Raise events on temperature & pressure
154  mpl_tx_data[1] = 0x07;
155  mpl_write_ctx.write_length = 2;
156  mpl_write_ctx.callback_pointer = &loc_mpl_callback_to_init2;
157  mpl_state = E_MPL_STATE_PENDING;
158  mpl_timeout_timer = hal_timer_getTimeout(TIMEOUT_DELAY);
159  i2c_bus_mgr_queue_write(mpl_i2c_node,&mpl_write_ctx);
160  break;
161  case E_MPL_STATE_INIT2:
162  mpl_tx_data[0] = MPL_REG_CTRL1;
163  // Set to Barometer with an OSR = 128 (standby off)
164  mpl_tx_data[1] = 0x39;
165  mpl_write_ctx.write_length = 2;
166  mpl_write_ctx.callback_pointer = &loc_mpl_callback_to_init_finish;
167  mpl_state = E_MPL_STATE_PENDING;
168  mpl_timeout_timer = hal_timer_getTimeout(TIMEOUT_DELAY);
169  i2c_bus_mgr_queue_write(mpl_i2c_node,&mpl_write_ctx);
170  break;
171  case E_MPL_STATE_IDLE:
172  mpl_tx_data[0] = 0x00;
173  mpl_read_ctx.write_length = 1;
174  mpl_rx_data[0] = 0x00;
175  mpl_read_ctx.read_length = 1;
176  mpl_read_ctx.callback_pointer = &loc_mpl_callback_to_ready;
177  mpl_state = E_MPL_STATE_PENDING;
178  mpl_timeout_timer = hal_timer_getTimeout(TIMEOUT_DELAY);
179  i2c_bus_mgr_queue_read(mpl_i2c_node,&mpl_read_ctx);
180  break;
181 
182  case E_MPL_STATE_PENDING:
183  if (hal_timer_isTimedOut(mpl_timeout_timer)) {
184  mpl_state = E_MPL_STATE_TIMEOUT;
185  }
186  break;
187  case E_MPL_STATE_READY:
188  mpl_tx_data[0] = 0x01;
189  mpl_read_ctx.write_length = 1;
190  mpl_rx_data[0] = 0x00;
191  mpl_rx_data[1] = 0x00;
192  mpl_rx_data[2] = 0x00;
193  mpl_rx_data[3] = 0x00;
194  mpl_rx_data[4] = 0x00;
195  mpl_read_ctx.read_length = 5;
196  mpl_read_ctx.callback_pointer = &loc_mpl_callback_to_idle;
197  mpl_state = E_MPL_STATE_PENDING;
198  mpl_timeout_timer = hal_timer_getTimeout(TIMEOUT_DELAY);
199  i2c_bus_mgr_queue_read(mpl_i2c_node,&mpl_read_ctx);
200  break;
201  case E_MPL_STATE_TIMEOUT:
202  break;
203  }
204 }
