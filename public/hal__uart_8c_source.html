194 static uint32_t loc_countTxDataBytesLinear(s_hal_uart_ctx_t *ps_ctx)
195 {
196  uint32_t l_return = 0;
197 
198  if( IS_BUF_EMPTY(ps_ctx->pc_txDataRead, ps_ctx->pc_txDataWrite) )
199  {
200  l_return = 0;
201  }
202  else if( ps_ctx->pc_txDataRead < ps_ctx->pc_txDataWrite)
203  {
204  l_return = ps_ctx->pc_txDataWrite - ps_ctx->pc_txDataRead;
205  }
206  else
207  {
208  l_return = &(ps_ctx->ac_txData[UART_TX_DATA_BUF_SIZE-1]) - ps_ctx->pc_txDataRead;
209  }
210 
211  return l_return;
212 }

214 static void loc_incrementBy(s_hal_uart_ctx_t *ps_ctx)
215 {
216  ps_ctx->pc_txDataRead += ps_ctx->txDataTransmitCount;
217  ps_ctx->txDataTransmitCount = 0;
218 
219  /* Ring buf! Therefore check and do wrap around if required. */
220  if( ps_ctx->pc_txDataRead == &ps_ctx->ac_txData[UART_TX_DATA_BUF_SIZE-1] )
221  {
222  ps_ctx->pc_txDataRead = ps_ctx->ac_txData;
223  }
224 }

226 static bool loc_isRxBufFull(s_hal_uart_ctx_t *ps_ctx)
227 {
228  bool b_return = false;
229 
230  /* buf is full if read pointer is 1 byte ahead the write pointer. */
231  if( (ps_ctx->pc_rxDataWrite + 1) == ps_ctx->pc_rxDataRead)
232  {
233  b_return = true;
234  }
235  /* Due to the ringbuf it might be, that the read pointer was already wrapped around.
236  * We must check with "SIZE-2" here, as the write pointer will wrap around at
237  * "SIZE-1" already and thereby point to start again. */
238  else if( (ps_ctx->pc_rxDataWrite == &(ps_ctx->ac_rxData[UART_RX_DATA_BUF_SIZE-2]) ) &&
239  (ps_ctx->pc_rxDataRead == &(ps_ctx->ac_rxData[0]) ) )
240  {
241  b_return = true;
242  }
243 
244  return b_return;
245 }

268 static bool loc_writeToTxFIFO(s_hal_uart_ctx_t *ps_ctx)
269 {
270  bool b_return = false;
271 
272  /* Verify that space if left in the MCU TX FIFO and our buffer is not empty */
273  while( MAP_UARTSpaceAvail(ps_ctx->ui32_baseAddr) &&
274  !IS_BUF_EMPTY(ps_ctx->pc_txDataWrite, ps_ctx->pc_txDataRead) )
275  {
276  MAP_UARTCharPutNonBlocking(ps_ctx->ui32_baseAddr, *ps_ctx->pc_txDataRead++);
277  b_return = true;
278 
279  /* Ring buf! Therefore check and do wrap around if required. */
280  if( ps_ctx->pc_txDataRead == &ps_ctx->ac_txData[UART_TX_DATA_BUF_SIZE-1] )
281  {
282  ps_ctx->pc_txDataRead = ps_ctx->ac_txData;
283  }
284  }/* while() */
285 
286  return b_return;
287 }

289 static bool loc_readByteFromBuf(s_hal_uart_ctx_t *ps_ctx, uint8_t *pc_byte)
290 {
291  bool b_return = false;
292 
293  if( !IS_BUF_EMPTY(ps_ctx->pc_rxDataWrite, ps_ctx->pc_rxDataRead) )
294  {
295  *pc_byte = *ps_ctx->pc_rxDataRead++;
296 
297  /* Ring buf! Therefore check and do wrap around if required. */
298  if( ps_ctx->pc_rxDataRead == &ps_ctx->ac_rxData[UART_RX_DATA_BUF_SIZE-1] )
299  {
300  ps_ctx->pc_rxDataRead = ps_ctx->ac_rxData;
301  }
302 
303  b_return = true;
304  }
305 
306  return b_return;
307 }

309 /* ===========================================================================*/
310 /* loc_handleInterrupt() */
311 /* ===========================================================================*/
312 #if !( HAL_UART0_USE_DMA & HAL_UART1_USE_DMA & HAL_UART2_USE_DMA & HAL_UART3_USE_DMA & \
313  HAL_UART4_USE_DMA & HAL_UART5_USE_DMA & HAL_UART7_USE_DMA )
314 static void loc_handleInterrupt(s_hal_uart_ctx_t *ps_ctx)
315 {
316  uint32_t ui32Status;
317 
318  //
319  // Get the interrrupt status.
320  //
321  ui32Status = MAP_UARTIntStatus(ps_ctx->ui32_baseAddr, true);
322 
323  //
324  // Clear the asserted interrupts.
325  //
326  MAP_UARTIntClear(ps_ctx->ui32_baseAddr, ui32Status);
327 
328  if( (ui32Status & UART_INT_RX) || (ui32Status & UART_INT_RT) )
329  {
330  //
331  // Loop while there are characters in the receive FIFO.
332  //
333  while(MAP_UARTCharsAvail(ps_ctx->ui32_baseAddr))
334  {
335  /* Only write into RX ringbuf if not full! Otherwise, the writePointer
336  * will overrun readPointer and hal_uart_read() may return old/outdated
337  * data again. This is seen more critical than missing/ommit the bytes we
338  * can't store anymore.
339  */
340  if( !loc_isRxBufFull(ps_ctx) )
341  {
342  *ps_ctx->pc_rxDataWrite++ = MAP_UARTCharGetNonBlocking(ps_ctx->ui32_baseAddr);
343 
344  /* Ring buf! Therefore check and do wrap around if required. */
345  if( ps_ctx->pc_rxDataWrite == &ps_ctx->ac_rxData[UART_RX_DATA_BUF_SIZE-1] )
346  {
347  ps_ctx->pc_rxDataWrite = ps_ctx->ac_rxData;
348  }
349  }
350  else
351  {
352  /* Break the while() here to not stay in ISR for inifinity. */
353  break;
354  }
355  }
356  }
357 
358  /* Check if TX interrupt occurred. */
359  if(ui32Status & UART_INT_TX)
360  {
361  if( IS_BUF_EMPTY(ps_ctx->pc_txDataWrite, ps_ctx->pc_txDataRead) )
362  {
363  /* No further data to send so disable TX interrupt. */
364  MAP_UARTIntDisable(ps_ctx->ui32_baseAddr, UART_INT_TX);
365  }
366  else
367  {
368  /* We still have data to write through the UART. */
369  loc_writeToTxFIFO(ps_ctx);
370  }/* if() ... else */
371  }
372 
373  return;
374 }
375 #endif

377 /* ===========================================================================*/
378 /* loc_handleDmaInterrupt() */
379 /* ===========================================================================*/
380 #if( HAL_UART0_USE_DMA | HAL_UART1_USE_DMA | HAL_UART2_USE_DMA | HAL_UART3_USE_DMA | \
381  HAL_UART4_USE_DMA | HAL_UART5_USE_DMA | HAL_UART7_USE_DMA )
382 static void loc_handleDmaInterrupt(s_hal_uart_ctx_t *ps_ctx)
383 {
384  uint32_t ui32Status = 0;
385 
386  //
387  // Get the interrrupt status.
388  //
389  ui32Status = MAP_UARTIntStatus(ps_ctx->ui32_baseAddr, true);
390 
391  //
392  // Clear the asserted interrupts.
393  //
394  MAP_UARTIntClear(ps_ctx->ui32_baseAddr, ui32Status);
395 
396  if( (ui32Status & UART_INT_RX) || (ui32Status & UART_INT_RT) )
397  {
398  //
399  // Loop while there are characters in the receive FIFO.
400  //
401  while(MAP_UARTCharsAvail(ps_ctx->ui32_baseAddr))
402  {
403  /* Only write into RX ringbuf if not full! Otherwise, the writePointer
404  * will overrun readPointer and hal_uart_read() may return old/outdated
405  * data again. This is seen more critical than missing/ommit the bytes we
406  * can't store anymore.
407  */
408  if( !loc_isRxBufFull(ps_ctx) )
409  {
410  *ps_ctx->pc_rxDataWrite++ = MAP_UARTCharGetNonBlocking(ps_ctx->ui32_baseAddr);
411 
412  /* Ring buf! Therefore check and do wrap around if required. */
413  if( ps_ctx->pc_rxDataWrite == &ps_ctx->ac_rxData[UART_RX_DATA_BUF_SIZE-1] )
414  {
415  ps_ctx->pc_rxDataWrite = ps_ctx->ac_rxData;
416  }
417  }
418  else
419  {
420  /* Break the while() here to not stay in ISR for inifinity. */
421  break;
422  }
423  }
424  }
425 
426  //
427  // If the UART1 DMA TX channel is disabled, that means the TX DMA transfer
428  // is done.
429  //
430  if( ui32Status & UART_INT_DMATX )
431  {
432  if(!MAP_uDMAChannelIsEnabled(ps_ctx->ui32_dmaChannelTx))
433  {
434  /* Increment pointers first. */
435  loc_incrementBy(ps_ctx);
436  ps_ctx->txDataTransmitCount = loc_countTxDataBytesLinear(ps_ctx);
437 
438  if( ps_ctx->txDataTransmitCount )
439  {
440  //
441  // Start another DMA transfer to UART TX.
442  //
443  MAP_uDMAChannelTransferSet(ps_ctx->ui32_dmaChannelTx | UDMA_PRI_SELECT,
444  UDMA_MODE_BASIC, (void *)ps_ctx->pc_txDataRead,
445  (void *)(ps_ctx->ui32_baseAddr + UART_O_DR),
446  ps_ctx->txDataTransmitCount
447  );
448 
449  //
450  // The uDMA TX channel must be re-enabled.
451  //
452  MAP_uDMAChannelEnable(ps_ctx->ui32_dmaChannelTx);
453  }
454  else
455  {
456  ps_ctx->txEnabled = false;
457  //
458  // Disable the UART DMA TX interrupt.
459  //
460  MAP_UARTIntDisable(ps_ctx->ui32_baseAddr, UART_INT_DMATX );
461  }
462  }
463  }
464 
465  return;
466 }
467 #endif

469 /* ===========================================================================*/
470 /* loc_initDmaTx() */
471 /* ===========================================================================*/
472 #if( HAL_UART0_USE_DMA | HAL_UART1_USE_DMA | HAL_UART2_USE_DMA | HAL_UART3_USE_DMA | \
473  HAL_UART4_USE_DMA | HAL_UART5_USE_DMA | HAL_UART7_USE_DMA )
474 static void loc_initDmaTx(s_hal_uart_ctx_t *ps_ctx, uint32_t ui32_dmaChanAssignTx)
475 {
476  //
477  // Enable the UART for operation, and enable the uDMA interface for TX.
478  //
479  MAP_UARTEnable(ps_ctx->ui32_baseAddr);
480  MAP_UARTDMAEnable(ps_ctx->ui32_baseAddr, UART_DMA_TX);
481  uDMAChannelEnable(ps_ctx->ui32_dmaChannelTx);
482  uDMAChannelAssign(ui32_dmaChanAssignTx);
483 
484  //
485  // Put the attributes in a known state for the uDMA UART1TX channel. These
486  // should already be disabled by default.
487  //
488  MAP_uDMAChannelAttributeDisable(ps_ctx->ui32_dmaChannelTx,
489  UDMA_ATTR_ALTSELECT |
490  UDMA_ATTR_HIGH_PRIORITY |
491  UDMA_ATTR_REQMASK);
492 
493  //
494  // Set the USEBURST attribute for the uDMA UART TX channel. This will
495  // force the controller to always use a burst when transferring data from
496  // the TX buffer to the UART. This is somewhat more effecient bus usage
497  // than the default which allows single or burst transfers.
498  //
499  MAP_uDMAChannelAttributeEnable(ps_ctx->ui32_dmaChannelTx, UDMA_ATTR_USEBURST);
500 
501  //
502  // Configure the control parameters for the UART TX. The uDMA UART TX
503  // channel is used to transfer a block of data from a buffer to the UART.
504  // The data size is 8 bits. The source address increment is 8-bit bytes
505  // since the data is coming from a buffer. The destination increment is
506  // none since the data is to be written to the UART data register. The
507  // arbitration size is set to 4, which matches the UART TX FIFO trigger
508  // threshold.
509  //
510  MAP_uDMAChannelControlSet(ps_ctx->ui32_dmaChannelTx | UDMA_PRI_SELECT,
511  UDMA_SIZE_8 | UDMA_SRC_INC_8 |
512  UDMA_DST_INC_NONE |
513  UDMA_ARB_4);
514 
515  return;
516 }
517 #endif

522 static void loc_initUartStruct(s_hal_uart_ctx_t *ps_ctx, uint32_t ui32_baseAddr, uint32_t ui32_dmaChannelTx)

528  ps_ctx->pc_txDataRead = ps_ctx->ac_txData;

532  ps_ctx->txEnabled = false;
533  return;
534 }

539 static s_hal_uart_ctx_t* loc_initUart0(uint32_t ui32_sysClock, uint32_t ui32_baudRate)
540 {
541  //
542  // Enable the peripherals used by this example.
543  //
544  MAP_SysCtlPeripheralEnable(SYSCTL_PERIPH_UART0);
545  MAP_SysCtlPeripheralEnable(SYSCTL_PERIPH_GPIOA);
546 
547  //
548  // Set GPIO A0 and A1 as UART pins.
549  //
550  GPIOPinConfigure(GPIO_PA0_U0RX);
551  GPIOPinConfigure(GPIO_PA1_U0TX);
552  MAP_GPIOPinTypeUART(GPIO_PORTA_BASE, GPIO_PIN_0 | GPIO_PIN_1);
553 
554  //
555  // Configure the UART for 8-N-1 operation with user defined baud rate.
556  //
557  MAP_UARTConfigSetExpClk(UART0_BASE, ui32_sysClock, ui32_baudRate,
558  (UART_CONFIG_WLEN_8 | UART_CONFIG_STOP_ONE |
559  UART_CONFIG_PAR_NONE));
560 
561  MAP_UARTFIFOLevelSet(UART0_BASE, UART_FIFO_TX4_8, UART_FIFO_RX4_8);
562 
563  /* Initialize our context struct. */
564  #if HAL_UART0_USE_DMA
565  loc_initUartStruct( &gs_uart0, UART0_BASE, UDMA_CH9_UART0TX );
566  loc_initDmaTx(&gs_uart0, UDMA_CH9_UART0TX);
567  #else
568  loc_initUartStruct( &gs_uart0, UART0_BASE, INVALID_DMA_CHANNEL );
569  #endif
570 
571  //
572  // Enable the UART interrupt.
573  //
574  UARTIntRegister( UART0_BASE, UART0IntHandler );
575  MAP_UARTIntEnable(UART0_BASE, UART_INT_RX | UART_INT_RT);
576  MAP_UARTIntDisable(UART0_BASE, UART_INT_TX);
577  MAP_IntEnable(INT_UART0);
578 
579  return &gs_uart0;
580 }

987 /* ===========================================================================*/
988 /* hal_uart_write() */
989 /* ===========================================================================*/
990 uint16_t hal_uart_write(s_hal_uart_ctx_t *ps_ctx, uint8_t *pc_data, uint16_t dataLen)
991 {
992  uint16_t i_ret = 0;
993 
994  if( ps_ctx && pc_data && (dataLen > 0) )
995  {
996  for (i_ret = 0; i_ret < dataLen; ++i_ret)
997  {
998  if(false == loc_isTxBufFull(ps_ctx))
999  {
1000  *ps_ctx->pc_txDataWrite++ = *pc_data++;
1001 
1002  /* Ring buf! Therefore check and do wrap around if required. */
1003  if( ps_ctx->pc_txDataWrite == &ps_ctx->ac_txData[UART_TX_DATA_BUF_SIZE-1] )
1004  {
1005  ps_ctx->pc_txDataWrite = ps_ctx->ac_txData;
1006  }
1007  }
1008  else
1009  {
1010  break; /* No further space left in TX buffer. */
1011  }
1012  }/* for() */
1013 
1014  if( ps_ctx->ui32_dmaChannelTx != INVALID_DMA_CHANNEL)
1015  {
1016  /* Make sure we wrote data to the TX buffer and that currently no
1017  * TX transmission is ongoing. */
1018  if( (i_ret > 0) && !ps_ctx->txEnabled )
1019  {
1020  ps_ctx->txEnabled = true;
1021 
1022  //
1023  // Enable the UART DMA TX interrupt.
1024  //
1025  MAP_UARTIntEnable(ps_ctx->ui32_baseAddr, UART_INT_DMATX );
1026  }
1027  }
1028  else
1029  {
1030  /*
1031  * This UART port does not use DMA!
1032  */
1033  if( (i_ret > 0) && ( !UARTBusy(ps_ctx->ui32_baseAddr) ) )
1034  {
1035 
1036  if( loc_writeToTxFIFO(ps_ctx) )
1037  {
1038  /* We need to trigger the TX ISR in case we wrote new data to the TX ring buf. */
1039  MAP_UARTIntEnable(ps_ctx->ui32_baseAddr, UART_INT_TX);
1040  }
1041  }
1042  }
1043  }
1044 
1045  return i_ret;
1046 }
