202 void XbusParser_parseByte(struct XbusParser* parser, const uint8_t byte)
203 {
204  switch (parser->state)
205  {
206  case XBPS_Preamble:
207  if (byte == XBUS_PREAMBLE)
208  {
209  parser->checksum = 0;
210  parser->state = XBPS_BusId;
211  }
212  break;
213 
214  case XBPS_BusId:
215  parser->checksum += byte;
216  parser->state = XBPS_MessageId;
217  break;
218 
219  case XBPS_MessageId:
220  parser->checksum += byte;
221  parser->currentMessage.mid = (enum XsMessageId)byte;
222  parser->state = XBPS_Length;
223  break;
224 
225  case XBPS_Length:
226  parser->checksum += byte;
227  if (byte == XBUS_NO_PAYLOAD)
228  {
229  parser->currentMessage.length = byte;
230  parser->currentMessage.data = NULL;
231  parser->state = XBPS_Checksum;
232  }
233  else if (byte < XBUS_EXTENDED_LENGTH)
234  {
235  parser->currentMessage.length = byte;
236  prepareForPayload(parser);
237  parser->state = XBPS_Payload;
238  }
239  else
240  {
241  parser->state = XBPS_ExtendedLengthMsb;
242  }
243  break;
244 
245  case XBPS_ExtendedLengthMsb:
246  parser->checksum += byte;
247  parser->currentMessage.length = ((uint16_t)byte) << 8;
248  parser->state = XBPS_ExtendedLengthLsb;
249  break;
250 
251  case XBPS_ExtendedLengthLsb:
252  parser->checksum += byte;
253  parser->currentMessage.length |= byte;
254  prepareForPayload(parser);
255  parser->state = XBPS_Payload;
256  break;
257 
258  case XBPS_Payload:
259  parser->checksum += byte;
260  if (parser->currentMessage.data)
261  {
262  ((uint8_t*)parser->currentMessage.data)[parser->payloadReceived] = byte;
263  }
264  if (++parser->payloadReceived == parser->currentMessage.length)
265  {
266  parser->state = XBPS_Checksum;
267  }
268  break;
269

271  parser->checksum += byte;
272  if ((parser->checksum == 0) &&
273  ((parser->currentMessage.length == 0) ||
274  parser->currentMessage.data))

278  }

282  }

284  break;
285  }
