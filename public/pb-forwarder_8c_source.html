137 bool pb_forwarder_init( bool (*pfnForward)(uint8_t *pc_data, uint16_t i_len) )
138 {
139  bool b_return = false;
140 
141  /* 1st, set OE high for TXS0108 Level Shifter. */
142  hal_gpio_setDir(MFW_LSHIFT_EN_PORT, MFW_LSHIFT_EN_PIN, E_HAL_GPIO_DIR_OUT);
143  hal_gpio_write(MFW_LSHIFT_EN_PORT, MFW_LSHIFT_EN_PIN, MFW_LSHIFT_EN_PIN);
144 
145  if(pfnForward)
146  {
147  /* Store the function pointer given to forward data. */
148  pfn_forwardData = pfnForward;
149 
150  /* Initialize the UART 1 to listen for data to be forwarded. */
151  gps_uart1Listener = hal_uart_init( E_HAL_UART_PORT_1, 921600 );
152  if(gps_uart1Listener != NULL)
153  {
154  b_return = true;
155  }
156 
157  /* Init the serial protocol module used for reading from UART 1. */
158  if( b_return )
159  {
160  b_return = sproto_init(&gs_sprotoListenerUart1, /* context */
161  loc_cbWriteToUart1, loc_cbReadFromUart1, loc_cbUart1RxDataRdy, /* callbacks */
162  loc_cbSerialToUart1Error, loc_cbSerialCrcCalculate,
163  gac_uart1RxBuf, sizeof(gac_uart1RxBuf) ); /* buffer */
164  }
165  }
166 
167  return b_return;
168 }
