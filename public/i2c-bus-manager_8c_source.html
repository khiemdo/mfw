52 void loc_i2c_bus_mgr_assign_access(){
53  if (g_current_access == E_I2C_BUS_NODE_NONE) {
54  uint8_t values[NUMBER_OF_NODES];
55  values[0] = g_node_priorities.node_sht25_prio;
56  values[1] = g_node_priorities.node_pressure_prio;
57  values[2] = g_node_priorities.node_led_drv_prio;
58  values[3] = g_node_priorities.node_adc1_prio;
59  values[4] = g_node_priorities.node_adc2_prio;
60  values[5] = g_node_priorities.node_adc3_prio;
61 
62  uint8_t min = 255;
63  uint8_t min_ind = NUMBER_OF_NODES;
64  uint8_t i;
65  for (i = 0; i < NUMBER_OF_NODES; i++) {
66  if (values[i] < min) {
67  min = values[i];
68  min_ind = i;
69  }
70  }
71 
72  switch (min_ind) {
73  case 0:
74  g_current_access = E_I2C_BUS_NODE_SHT25;
75  break;
76  case 1:
77  g_current_access = E_I2C_BUS_NODE_PRESSURE;
78  break;
79  case 2:
80  g_current_access = E_I2C_BUS_NODE_LED_DRV;
81  break;
82  case 3:
83  g_current_access = E_I2C_BUS_NODE_ADC1;
84  break;
85  case 4:
86  g_current_access = E_I2C_BUS_NODE_ADC2;
87  break;
88  case 5:
89  g_current_access = E_I2C_BUS_NODE_ADC3;
90  break;
91  case NUMBER_OF_NODES:
92  g_current_access = E_I2C_BUS_NODE_NONE;
93  break;
94  }
95 
96  }
97 }
