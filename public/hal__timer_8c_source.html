59 bool hal_timer_init(void)
60 {

62  // Configure SysTick to occur 1000 times per second, to use as a time
63  // reference. Enable SysTick to generate interrupts.
64  //
65  SysTickIntRegister( SysTickHandler );

94 bool hal_timer_registerCallback(uint32_t intervall, void (*pfn_callback)(void))
95 {
96  bool b_return = false;
97 
98  if(pfn_callback)
99  {
100  g_timer.pfn_callback = pfn_callback;
101  g_timer.intervall = intervall;
102  /* Immediately start the timer. */

104  b_return = true;

110 //*****************************************************************************
111 //
112 // The interrupt handler for the SysTick timer. This handler will increment a
113 // milliseconds counter in each tick (in each call respectively.
114 //
115 //*****************************************************************************
116 void
117 SysTickHandler(void)
118 {
119  //
120  // Increment the tick counter.
121  //
122  g_ui32TickCount++;
123 
124  /* A single timer with callback can be requested by this module.
125  * Serve it if present/requested. */
126  if( g_timer.pfn_callback )
127  {
128  if( g_ui32TickCount >= g_timer.timeout )
129  {
130  g_timer.timeout += g_timer.intervall;
131  g_timer.pfn_callback();
132  }
133  }
134 
135  return;
136 }
