41 bool servopower_status(void)
42 {
43  bool b_return = false;
44  uint8_t ui8_status = 0;
45  ui8_status = hal_gpio_read(E_HAL_GPIO_PORT_N, HAL_GPIO_PIN_3);
46  /* If pin is low, power is on. */
47  if(!(ui8_status & HAL_GPIO_PIN_3))
48  {
49  b_return = true;
50  }
51  /* If pin is high, power is off. */
52  else
53  {
54  b_return = false;
55  }
56  return b_return;

58 }
