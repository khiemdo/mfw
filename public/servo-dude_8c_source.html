222 static bool loc_init_i2c_handler(E_SERVO_DUDE_ID_t e_servo_dude_id)
223 {
224  bool b_return = true;
225  switch(e_servo_dude_id)
226  {
227  case(SD_I2C_0_PORT_ID1):
228  i2c_0.has_turn = e_servo_dude_id;
229  i2c_0.id1 = e_servo_dude_id;
230  break;
231  case(SD_I2C_0_PORT_ID2):
232  i2c_0.has_turn = e_servo_dude_id;
233  i2c_0.id2 = e_servo_dude_id;
234  break;
235  case(SD_I2C_1_PORT_ID1):
236  i2c_1.has_turn = e_servo_dude_id;
237  i2c_1.id1 = e_servo_dude_id;
238  break;
239  case(SD_I2C_1_PORT_ID2):
240  i2c_1.has_turn = e_servo_dude_id;
241  i2c_1.id2 = e_servo_dude_id;
242  break;
243  case(SD_I2C_2_PORT_ID1):
244  i2c_2.has_turn = e_servo_dude_id;
245  i2c_2.id1 = e_servo_dude_id;
246  break;
247  case(SD_I2C_2_PORT_ID2):
248  i2c_2.has_turn = e_servo_dude_id;
249  i2c_2.id2 = e_servo_dude_id;
250  break;
251  default:
252  b_return = false;
253  }
254  return b_return;
255 }

457 /* servo_dude_init() Initializes a ServoDude and returns context */
458 /* ===========================================================================*/
459 s_servo_dude_ctx_t *servo_dude_init(E_SERVO_DUDE_ID_t e_servo_dude_id)
460 {
461  loc_init_i2c_handler(e_servo_dude_id);
462  s_servo_dude_ctx_t *ps_return = NULL;
463  switch(e_servo_dude_id)
464  {
465  case E_SERVO_DUDE_ELEVATOR:
466  ps_return = loc_init_sd_el();
467  break;
468  case E_SERVO_DUDE_RUDDER:
469  ps_return = loc_init_sd_rd();
470  break;
471  case E_SERVO_DUDE_FLAP_LEFT:
472  ps_return = loc_init_sd_fl();
473  break;
474  case E_SERVO_DUDE_FLAP_RIGHT:
475  ps_return = loc_init_sd_fr();
476  break;
477  case E_SERVO_DUDE_AILERON_LEFT:
478  ps_return = loc_init_sd_al();
479  break;
480  case E_SERVO_DUDE_AILERON_RIGHT:
481  ps_return = loc_init_sd_ar();
482  break;
483  /* Other ServoDude's are not supported => ps_return remains NULL. */
484  default:
485  break;
486  }
487 
488  return ps_return;
489 }

491 /* ===========================================================================*/
492 /* loc_handle_limits() - Sets the limits of the specified servo dude */
493 /* ===========================================================================*/
494 void loc_handle_limits(s_servo_dude_ctx_t *const servodude_ctx)
495 {
496  if((servodude_ctx->max_limit_av) && (servodude_ctx->state == E_SERVO_DUDE_STATE_IDLE))
497  {
498  uint8_t max_lim[2];
499  max_lim[0] = (servodude_ctx->max_limit >> 8) | 0x80;
500  max_lim[1] = servodude_ctx->max_limit & 0xFF;
501  servodude_ctx->max_limit_av = false;
502  servodude_ctx->state = E_SERVO_DUDE_STATE_PENDING;
503  switch(servodude_ctx->servo_id)
504  {
505  case E_SERVO_DUDE_ELEVATOR:
506  hal_i2c_write(servodude_ctx->i2c_ctx, servodude_ctx->slave_address, max_lim, 2, &loc_el_callback);
507  break;
508  case E_SERVO_DUDE_RUDDER:
509  hal_i2c_write(servodude_ctx->i2c_ctx, servodude_ctx->slave_address, max_lim, 2, &loc_rd_callback);
510  break;
511  case E_SERVO_DUDE_FLAP_LEFT:
512  hal_i2c_write(servodude_ctx->i2c_ctx, servodude_ctx->slave_address, max_lim, 2, &loc_fl_callback);
513  break;
514  case E_SERVO_DUDE_FLAP_RIGHT:
515  hal_i2c_write(servodude_ctx->i2c_ctx, servodude_ctx->slave_address, max_lim, 2, &loc_fr_callback);
516  break;
517  case E_SERVO_DUDE_AILERON_LEFT:
518  hal_i2c_write(servodude_ctx->i2c_ctx, servodude_ctx->slave_address, max_lim, 2, &loc_al_callback);
519  break;
520  case E_SERVO_DUDE_AILERON_RIGHT:
521  hal_i2c_write(servodude_ctx->i2c_ctx, servodude_ctx->slave_address, max_lim, 2, &loc_ar_callback);
522  break;
523  }
524  }
525  if((servodude_ctx->min_limit_av) && (servodude_ctx->state == E_SERVO_DUDE_STATE_IDLE))
526  {
527  uint8_t min_lim[2];
528  min_lim[0] = (servodude_ctx->min_limit >> 8) | 0xC0;
529  min_lim[1] = servodude_ctx->min_limit & 0xFF;
530  servodude_ctx->min_limit_av = false;
531  servodude_ctx->state = E_SERVO_DUDE_STATE_PENDING;
532  switch(servodude_ctx->servo_id)
533  {
534  case E_SERVO_DUDE_ELEVATOR:
535  hal_i2c_write(servodude_ctx->i2c_ctx, servodude_ctx->slave_address, min_lim, 2, &loc_el_callback);
536  break;
537  case E_SERVO_DUDE_RUDDER:
538  hal_i2c_write(servodude_ctx->i2c_ctx, servodude_ctx->slave_address, min_lim, 2, &loc_rd_callback);
539  break;
540  case E_SERVO_DUDE_FLAP_LEFT:
541  hal_i2c_write(servodude_ctx->i2c_ctx, servodude_ctx->slave_address, min_lim, 2, &loc_fl_callback);
542  break;
543  case E_SERVO_DUDE_FLAP_RIGHT:
544  hal_i2c_write(servodude_ctx->i2c_ctx, servodude_ctx->slave_address, min_lim, 2, &loc_fr_callback);
545  break;
546  case E_SERVO_DUDE_AILERON_LEFT:
547  hal_i2c_write(servodude_ctx->i2c_ctx, servodude_ctx->slave_address, min_lim, 2, &loc_al_callback);
548  break;
549  case E_SERVO_DUDE_AILERON_RIGHT:
550  hal_i2c_write(servodude_ctx->i2c_ctx, servodude_ctx->slave_address, min_lim, 2, &loc_ar_callback);
551  break;
552  }
553  }
554  if((servodude_ctx->neutral_av) && (servodude_ctx->state == E_SERVO_DUDE_STATE_IDLE))
555  {
556  uint8_t neutral[2];
557  neutral[0] = (servodude_ctx->neutral >> 8) | 0x40;
558  neutral[1] = servodude_ctx->neutral & 0xFF;
559  servodude_ctx->neutral_av = false;
