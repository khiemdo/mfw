76 bool hal_eth_init(void)
77 {
78  bool b_ret = true;
79  uint32_t ui32User0 = 0U;
80  uint32_t ui32User1 = 0U;
81 
82  //
83  // Make sure the main oscillator is enabled because this is required by
84  // the PHY. The system must have a 25MHz crystal attached to the OSC
85  // pins. The SYSCTL_MOSC_HIGHFREQ parameter is used when the crystal
86  // frequency is 10MHz or higher.
87  //
88  SysCtlMOSCConfigSet(SYSCTL_MOSC_HIGHFREQ);
89 
90  /* Init ETH LEDs */
91  ROM_SysCtlPeripheralEnable(SYSCTL_PERIPH_GPIOF);
92  ROM_GPIOPinConfigure(GPIO_PF0_EN0LED0);
93  ROM_GPIOPinConfigure(GPIO_PF4_EN0LED1);
94 
95  GPIOPinTypeEthernetLED(GPIO_PORTF_BASE, GPIO_PIN_0 | GPIO_PIN_4);
96 
97 
98  /* Get and store pre-programmed MAC adress */
99  MAP_FlashUserSet(0xACABBEEF, 0xACABBEEF);
100  MAP_FlashUserGet(&ui32User0, &ui32User1);
101  if((ui32User0 == 0xffffffff) || (ui32User1 == 0xffffffff))
102  {
103  //
104  // We should never get here. This is an error if the MAC address has
105  // not been programmed into the device. Exit the program.
106  // Let the user know there is no MAC address
107  //
108  b_ret = false;
109  }
110  else
111  {
112  //
113  // Convert the 24/24 split MAC address from NV ram into a 32/16 split MAC
114  // address needed to program the hardware registers, then program the MAC
115  // address into the Ethernet Controller registers.
116  //
117  g_macAddress[0] = ((ui32User0 >> 0) & 0xff);
118  g_macAddress[1] = ((ui32User0 >> 8) & 0xff);
119  g_macAddress[2] = ((ui32User0 >> 16) & 0xff);
120  g_macAddress[3] = ((ui32User1 >> 0) & 0xff);
121  g_macAddress[4] = ((ui32User1 >> 8) & 0xff);
122  g_macAddress[5] = ((ui32User1 >> 16) & 0xff);
123  }
124 
125  if(b_ret)
126  {
127  /* TI demo sets priorities after low level initialization of ETH MAC/PHY
128  * but this seems to be working fine too. */
129 
130  //
131  // Set the interrupt priorities. We set the SysTick interrupt to a higher
132  // priority than the Ethernet interrupt to ensure that the file system
133  // tick is processed if SysTick occurs while the Ethernet handler is being
134  // processed. This is very likely since all the TCP/IP and HTTP work is
135  // done in the context of the Ethernet interrupt.
136  //
137  MAP_IntPrioritySet(INT_EMAC0, ETHERNET_INT_PRIORITY);
138  MAP_IntPrioritySet(FAULT_SYSTICK, SYSTICK_INT_PRIORITY);
139 
140 
141  /* Finally, mark that the module is initialized properly. */
142  g_isEthernetInitialized = true;
143  }
144 
145  return b_ret;
146 }
