55 bool hal_crc_init(void)
56 {
57  bool b_return = false;
58 
59  if (SysCtlPeripheralPresent(SYSCTL_PERIPH_CCM0))
60  {
61  //
62  // Enable the CRC module.
63  //
64  SysCtlPeripheralEnable(SYSCTL_PERIPH_CCM0);
65 
66  //
67  // Wait for the CRC module to be ready.
68  //
69  while(!SysCtlPeripheralReady(SYSCTL_PERIPH_CCM0));
70 
71  //
72  // Configure the CRC module.
73  //
74  CRCConfigSet(CCM0_BASE,
75  CRC_CFG_INIT_SEED |
76  CRC_CFG_TYPE_P1021 |
77  CRC_CFG_SIZE_8BIT);
78  g_hw_crc_av = true;
79  b_return = true;
80  }
81  else
82  {
83  // Use software crc
84  g_hw_crc_av = false;
85  b_return = true;
86  }
87  return b_return;
88 }

91 uint16_t hal_crc_calculate(const uint8_t *pc_data, uint16_t ui16_len)
92 {
93  uint16_t crc_result = 0;
94 
95  if(g_hw_crc_av)
96  {
97  /* Seed must be set for each new CRC calculation! */
98  CRCSeedSet(CCM0_BASE, HAL_CRC_CCITT_SEED);
99  /* Enable post-operation to finalize the CRC calculation within this single
100  * call. If process() shall be called several times for one CRC calculation
101  * this value shall be set to false for the first calls. */
102  crc_result = CRCDataProcess(CCM0_BASE, (uint32_t *) pc_data, ui16_len, true);
103  }
104  else
105  {
106  /* If no hardware CRC module is available, use software CRC for API functionality. */
107  crc_result = crcFast(pc_data, (uint32_t) ui16_len);
108  }
109 
110  return crc_result;
111 }
