93 /* ===========================================================================*/
94 /* loc_udpRx() */
95 /* ===========================================================================*/
96 static void
97 loc_udpRx(void *arg, struct udp_pcb *pcb, struct pbuf *p,
98  struct ip_addr *addr, u16_t port)
99 {
100  if(false == gb_dataAvailable)
101  {
102  /* Validate data sizes first. */
103  if(p->tot_len > g_dataBufSize)
104  {
105  g_lastDataLenRvcd = g_dataBufSize;
106  }
107  else
108  {
109  g_lastDataLenRvcd = p->tot_len;
110  }
111 
112  /* As this function loc_udpRx() is called from ETH interrupt we copy the data
113  * to user defined data buffer and raise the RX event in mfw_socket_udpRun()
114  * to make sure that the pb_decode() is not performed inside the interrupt. */
115  if( pbuf_copy_partial(p, g_dataBuf, g_lastDataLenRvcd, 0U) == g_lastDataLenRvcd )
116  {
117  gb_dataAvailable = true;
118  }
119  else
120  {
121  g_lastDataLenRvcd = 0U;
122  }
123  }
124  else
125  {
126  /* New data received while the last data is not yet processed by the user. */
127  g_errorDataOverflow++;
128  }
129 
130  /* The incoming pbuf is no longer needed, so free it. */
131  pbuf_free(p);
132 
133  return;
134 }

143 bool mfw_socket_init( uint32_t ui32IPAddr,
144  uint32_t ui32NetMask,
145  uint32_t ui32GWAddr
146  )
147 {
148  bool b_return = false;
149  const uint8_t *p_macAddr = NULL;
150  uint32_t sysClock = 0U;
151 
152  /* Proceed with initialization from bottom to top:
153  * 1) Init ethernet MAC/PHY. */
154  b_return = hal_eth_init();
155 
156  if(b_return)
157  {
158  p_macAddr = hal_eth_getMacAddress();
159  sysClock = hal_mcu_getSysClock();
160 
161  if(p_macAddr)
162  {
163  /* 2) Init lwIP stack, i.e. IP and socket (TCP, UDP) layer */
164  lwIPInit( sysClock,
165  p_macAddr,
166  ui32IPAddr,
167  ui32NetMask,
168  ui32GWAddr,
169  IPADDR_USE_STATIC
170  );
171 
172  /* Start a timer that periodically calls the ETH interrupt handler
173  * and thereby processes the lwIP stack. */
174  b_return = hal_timer_registerCallback(LWIP_TICK_MS, loc_lwIpTimerCallback);
175 
176  /* Init local variables here to ensure proper working if run() is called
177  * before udpStart(). */
178  g_errorDataOverflow = 0U;
179  gb_dataAvailable = false;
180 
181  g_dataBuf = NULL;
182  g_dataBufSize = 0U;
183  pfn_cb = NULL;
184  ip_addr_set_zero(&g_destIPAddr);
185  g_destPort = 0U;
186  pcb = NULL;
187  }
188 
189  }
190 
191  return b_return;
192 }

197 bool mfw_socket_udpStart( uint16_t srcPort,
198  uint32_t destIPAddr,
199  uint16_t destPort,
200  uint8_t *pc_rxDataBuf,
201  uint16_t rxDataBufLen,

203  )
204 {
205  bool b_return = false;
206  err_t err;
207 
208  /* Setup UDP port to communicate with GS. */
209  pcb = udp_new();
210  if(pcb)

212  udp_recv(pcb, loc_udpRx, NULL);
213 
214  /* Convert user param from host to network byte order. */
215  g_destIPAddr.addr = htonl(destIPAddr);
216 
217  err = udp_bind(pcb, IP_ADDR_ANY, srcPort);
218 
219  if(ERR_OK == err)
220  {
221  /* Store user parameter */
222  g_dataBuf = pc_rxDataBuf;
223  g_dataBufSize = rxDataBufLen;
224  memset(g_dataBuf, 0U, g_dataBufSize);
225  pfn_cb = callbackRxDataRdy;
226 
227  g_destPort = destPort;
228 
229  b_return = true;
230  }
231  else
232  {
233  ip_addr_set_zero(&g_destIPAddr);
234  }
235  }
236

241 /* mfw_socket_udpRun() */
242 /* ===========================================================================*/
243 void mfw_socket_udpRun(void)
244 {
245  //
246  // All the lwIP work is done in interrupt handlers. But we check if we
247  // received new data (within interrupt) that must be passed to user now.
248  //
249  if(gb_dataAvailable)
250  {
251  if(pfn_cb)
252  {
253  /* Inform user of received data and clear right afterwards. */
254  pfn_cb(g_dataBuf, g_lastDataLenRvcd);
255  memset(g_dataBuf, 0U, g_dataBufSize);
256  gb_dataAvailable = false;
257  }
258  }
259 
260  return;
261 }

266 bool mfw_socket_udpSend(uint8_t *data, uint16_t dataLen)
267 {
268  bool b_return = false;
269  struct pbuf *p = NULL;
270  err_t err = ERR_OK;
271 
272  /* Check for pcb to ensure udpStart() was called before. */
273  if(data && pcb)
274  {
275  /* Allocate a new pbuf for sending. */
276  p = pbuf_alloc(PBUF_TRANSPORT, dataLen, PBUF_RAM);
277 
278  if(p)
279  {
280  /* Copy the data into the pbuf, i.e. to lwIP stack memory. */
281  if( ERR_OK == pbuf_take(p, data, dataLen) )
282  {
283  /* Send the response. */
284  err = udp_sendto(pcb, p, &g_destIPAddr, g_destPort);
285  if( ERR_OK == err )
286  {
287  b_return = true;
288  }
289  }
290  }
291 
292  /* Free the pbuf. */
293  pbuf_free(p);
294  }
295 
296  return b_return;
297 }
